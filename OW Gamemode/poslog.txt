rule("Pos. Log With Player Info")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	conditions
	{
		Is Game In Progress == True;
		Is Moving(Event Player) == True;
	}

	actions
	{
		Log To Inspector(Custom String("{0}, {1}", Local Vector Of(Vector(0, 0, 0), Event Player, Rotation And Translation), Custom String(
			"{0}, {1}, {2}", Event Player, Team Of(Event Player), Hero Of(Event Player))));
		Wait(0.100, Abort When False);
		Loop If Condition Is True;
	}
}
